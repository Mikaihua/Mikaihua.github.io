6.新年倒计时，代码如下：
var boxa=document.getElementById('box');
			function countTime(){
				var time=new Date();
				var timeb=new Date("2017-01-01 00:00");
				var result=timeb-time;
				var day=parseInt(result/1000/3600/24);
				var hour=parseInt(result/1000/3600%24);
				var min=parseInt(result/1000/60%60);
				var sec=parseInt(result/1000%60);
				boxa.innerHTML="距离2017年还有"+day+"天"+hour+"时"+min+"分"+sec+"秒";
				setTimeout(countTime,1)
			}
			countTime();

运行结果是：距离2017年还有20天2时4分43秒
