一、this的指向问题

1.全局指向Global--window--Bom--Brower Object Model

（1）函数直接在全局调用时，this指向Window

例如

var num = 10;
function foo(){
	this.num = 1;
}
foo();
alert(num);
// 结果是弹出1 在window调用 相当于又声明一个 num = 1；

2、函数指向object

//指向 obj、弹出5
var num = 3 ;
var obj = {};
obj.num = 5;
obj.f = function () {
alert(this.num);
}
obj.f();

3、函数指向返回

var num = 10 ;
 function fun(){
return function(){
return this.num ;
}
	}
console.log(fun()());
//结果是10.

4、标签属性this指向当前标签指向

	< p  id = “p”   > 111< /p    >
	p.onclick=function(){console.log(this.innertext)}
	结果是111

5、Apply改变this的指向

	var  o = {num:400}
	var num = 10;
	var obj = {};
	obj.num = 1;
	obj.f = function(){
		alert(this.num)
	}
	obj.f.apply(); // 弹出10    如果第一个参数为空，函数内部的this指向window
	obj.f.apply(o); // 弹出400 向 o 添加obj.f的方法。
