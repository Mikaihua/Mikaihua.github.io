一、如何用cavas画猫

//首先，创建canvas画布
<canvas id="canvas" width="800" height="800">
//以下代码在script中
var canvas = document.getElementById("canvas");
var ctx = canvas.getContext('2d');
//脑袋下半部分
ctx.strokeStyle = "#34180C"                   //设置笔触颜色
ctx.lineWidth="5";                            //线条宽度
ctx.fillStyle = "#FFFFD9";                    //填充颜色 
ctx.beginPath();                              //起始路径
ctx.moveTo(180,300);                          //起始点
ctx.bezierCurveTo(200,420,490,420,500,290);   //两个控制点，一个结束点
ctx.fill();                                   //填充区域
ctx.stroke();                                 //绘制路径
//脑袋下半部画好了 。 同理画其他贝塞尔曲线。
 
//再画一根胡须，胡须是直线，比较好画
ctx.lineWidth="4";
ctx.moveTo(185,265);                          //从这个点
ctx.lineTo(170,265);                          //到这个点
ctx.stroke();                                 //闭合，一根胡须就好了 

//画个圆的，
//左眼睛亮点
ctx.fillStyle = "#fff";
ctx.beginPath();
ctx.arc(272,240,4,0,2*Math.PI);
ctx.closePath();
ctx.fill();
//其他的部分可依次画出，主要由贝塞尔曲线，弧度，圆，直线组成，再填充即可。

二、canvas载入图像

确定旋转中心点为画布中心点旋转得到。 
	








var left = document.getElementById("left");
var right = document.getElementById("right");
var c = document.getElementById("canvas");
var ctx = c.getContext("2d");

var img = new Image();
 img.onload = function(){
 	ctx.drawImage(img,-48,-64, 96,128);
 }
img.src = "./img/3.png";
ctx.translate((c.width/2),c.height/2);
left.addEventListener("click",turnLeft,false);
function turnLeft(){
	ctx.clearRect(-48,-64, 96,128);
	
ctx.rotate( 90*Math.PI/180);
ctx.drawImage(img,-48,-64, 96,128);
}

right.addEventListener("click",turnRight,false);
function turnRight(){
ctx.clearRect(-48,-64, 96,128)
ctx.rotate( -90*Math.PI/180);
ctx.drawImage(img,-48,-64, 96,128);
}
